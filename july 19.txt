package oit;
import java.util.*;
 
class Student
{
	Scanner sc=new Scanner(System.in);
	int sid;
	String sname;
	double per;
	void accept()
	{
		System.out.println("enter the student rollno, student name, percentage:");
		sid=sc.nextInt();
		
		sname=sc.next();
	
		per=sc.nextInt();
	}
	void display()
	{
		System.out.println("Student name:"+sname+"\nRollno:"+sid+"\npercentage:"+per);
	}
}

class july19
{
	public static void main(String args[])
	{
		Scanner sc=new Scanner(System.in);
		System.out.println("enter the Student array size:");
		int n=sc.nextInt();
		Student a[]=new Student[n];
		for(int i=0;i<n;i++)
		{
			a[i]=new Student();
			a[i].accept();
			a[i].display();
		}
	}
}
