package oit;
import java.util.*;

abstract class Employee3
{
	int eid;
	String ename;
	double bsal;
	
	Employee3()
	{
		eid=0;
		ename="x";
		bsal=0;
	}
	
	Employee3(int eid,String ename,double bsal)
	{
		this.eid=eid;
		this.ename=ename;
		this.bsal=bsal;
		
	}
	
	public void display()
	{
		System.out.println("Name:"+ename+"\nId:"+eid+"\nBasic salary:"+bsal);
	}
	abstract double  cal();
	
}

class fulltimeEmployee extends Employee3
{
	
	double ta,da,hra,netsal;
	fulltimeEmployee()
	{
		ta=0;
		da=0;
		hra=0;
	}
	
	fulltimeEmployee(int eid,String ename,double bsal,double ta,double da,double hra)
	{ super(eid,ename,bsal);
		this.ta=ta;
		this.da=da;
		this.hra=hra;
	}
	
	double cal()
	{
		netsal=bsal+(bsal*ta)+(bsal*da)+(bsal*hra);
		return netsal;
	}
    public  void display()
    {
    	System.out.println("Name:"+ename+"\nId:"+eid+"\nBasic salary:"+bsal+"\nta="+(bsal*ta)+"\nda="+(bsal*da)+"\nhra="+(bsal*hra));
    }
}

class parttimeEmployee extends Employee3
{
	double ta,perhour,hour,netsal;
	parttimeEmployee()
	{
		ta=0;
		perhour=0;
		hour=0;
	}
	parttimeEmployee(int eid,String ename,double bsal,double ta,double perhour,double hour)
	{ super(eid,ename,bsal);
		this.ta=ta;
		this.perhour=perhour;
		this.hour=hour;
	}
	double cal()
	{
		netsal=(perhour*hour*ta)+perhour*hour;
		return netsal;
	}
	 public  void display()
	    {
	    	System.out.println("Name:"+ename+"\nId:"+eid+"\nbasic salary="+(perhour*hour)+"\nta="+(perhour*hour*ta));
	    }
}

interface Printer
{
	void accept();
	void print();
}
abstract class shape
{
	abstract void print();
	
}
abstract class rectangle
{
	abstract void print();
	
}

class Employee4 implements Printer
{Scanner sc=new Scanner(System.in);
	int id;
	String name;
	double sal;
	
	public void accept()
	{
		System.out.println("Enter the id,name,salary:");
		id=sc.nextInt();
		name=sc.next();
		sal=sc.nextDouble();
	}
	
	public void print()
	{
		System.out.println("Id:"+id+"\nName:"+name+"\nsalary:"+sal);
	}
	
}
